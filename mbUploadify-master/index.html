mbUploadify html5移动端文件上传插件 (php)

案例一

<input type="file" name="files"  id="j-file" multiple>

<script>
    var upload = new mbUploadify({
        /*input file控件对象*/
        file: document.getElementById('j-file'),
        /*ajax 上传地址*/
        url: 'mbUploadify.php',
        /*ajax 提交的字段名称*/
        uploadName: '',

        /*上传成功*/
        load: function(e){
            document.getElementById('result').innerHTML += '<img src="'+ e.target.result +'">';
        },
        /*上传完成，不管成功失败*/
        loadend: function(){
            console.log('loadend!');
        },
        /*ajax上传成功*/
        uploadSuccess: function(res){
            console.log(res);
        },
        /*ajax上传失败*/
        uploadFailed: function(){
            console.log('upload failed!');
        },
        ......
    });
</script>

案例二

<form action="mbUploadify.php" method="post">
    <label for="j-file" class="mbUploadify" id="j-dropArea">
        文件上传
        <input type="file" name="files"  id="j-file"  multiple>
    </label>
    <!--   other   -->
    <input type="hidden" name="email" value="506713930@qq.com">
</form>
</pre>

<script>
    //传入form对象, 可附带其它数据一并提交
    var upload = new mbUploadify(document.querySelector('form'), {
        ...
        /*最多上传文件个数*/
        maximum: true,
        /*文件拖拽上传区域对象 null表示不支持*/
        dropElement: document.getElementById('j-dropArea'),
        ...
    });
</script>
